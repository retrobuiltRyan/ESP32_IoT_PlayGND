#include <Wire.h>
#include <Adafruit_PWMServoDriver.h>

Adafruit_PWMServoDriver pwm = Adafruit_PWMServoDriver();  // Default I2C address 0x40

// Servo pulse range for Tower Pro SG90 (approximate)
#define SERVOMIN  150   // ~500 microseconds
#define SERVOMAX  600   // ~2400 microseconds
#define SERVO_FREQ 50   // Standard analog servo frequency (50Hz)

void setup() {
  Serial.begin(115200);
  Serial.println("Tower Pro SG90 Servo Test with PCA9685");

  pwm.begin();
  pwm.setOscillatorFrequency(27000000);  // Recommended for accuracy
  pwm.setPWMFreq(SERVO_FREQ);            // Must be set correctly

  delay(10);  // Small delay after initializing
}

void loop() {
  int servoChannel = 0;  // Using channel 0 on PCA9685

  // Sweep from minimum to maximum
  for (uint16_t pulselen = SERVOMIN; pulselen <= SERVOMAX; pulselen += 5) {
    pwm.setPWM(servoChannel, 0, pulselen);
    delay(10);  // Smooth movement
  }

  delay(500);

  // Sweep back from maximum to minimum
  for (uint16_t pulselen = SERVOMAX; pulselen >= SERVOMIN; pulselen -= 5) {
    pwm.setPWM(servoChannel, 0, pulselen);
    delay(10);
  }

  delay(500);
}
